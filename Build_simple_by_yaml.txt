name: Build template

on:
  push:
    branches: [ main, master ]
    paths:
      - 'python_ai/**'
      - '.github/workflows/python_ai.yml'
  # pull_request:
  #   branches: [ main, master ]
  #   paths:
  #     - 'python_ai/**'
  #     - '.github/workflows/python_ai.yml'

jobs:
  build:
    runs-on: ubuntu-latest

    steps:
      - uses: actions/checkout@v4

      - name: Setup Python
        uses: actions/setup-python@v5
        with:
          python-version: '3.11'
          cache: 'pip'

      - name: Install dependencies
        run: |
          cd python_ai
          pip install --upgrade pip
          pip install -r requirements-dev.txt

      # (Tùy chọn) Chạy test nếu bạn có
      # - name: Run tests
      #   run: |
      #     cd python_ai
      #     python -m pytest   # hoặc python -m unittest

      - name: Upload artifact
        uses: actions/upload-artifact@v4
        with:
          name: python_ai-build
          path: python_ai/dist/*
          retention-days: 30    # Giữ artifact 30 ngày
          # Nếu muốn không nén (tùy chọn mới 2026): 
          # compression-level: 0
